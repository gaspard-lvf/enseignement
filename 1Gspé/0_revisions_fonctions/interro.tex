\documentclass{article}

\usepackage{expl3}
\usepackage{xparse}
\usepackage{pgffor}

\ExplSyntaxOn

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% 1. Création d'un dictionnaire (property list)
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\prop_new:N \g_equation_prop

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% 2. On remplit le dictionnaire
%%
%% Clé -> valeur
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\prop_put:Nnn \g_equation_prop {1}
{
\[
2x+3=0
\]
}

\prop_put:Nnn \g_equation_prop {2}
{
\[
-4x+4=4
\]
}

\prop_put:Nnn \g_equation_prop {3}
{
\[
12x+13=14
\]
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% 3. Variable contenant le numéro de version
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\int_new:N \g_version_int

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% 4. Commande utilisateur
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\NewDocumentCommand{\SetVersion}{m}
{
    \int_gset:Nn \g_version_int {#1}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% 5. Fonction interne
%%
%% Elle récupère la valeur du dictionnaire correspondant
%% à la version courante.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%N N e
%│ │ │
%│ │ └── troisième argument : expansion complète
%│ └──── deuxième argument : normal
%└────── premier argument : normal

%| Lettre | Signification                                |
%| ------ | -------------------------------------------- |
%| `N`    | nom d'une commande (`\ma_variable`)          |
%| `n`    | argument normal, sans expansion              |
%| `e`    | expansion complète (nouveau mécanisme expl3) |
%| `x`    | expansion complète classique                 |

\cs_new:Npn \equation_show:
{
    % On développe le numéro de version avant l'appel
    \exp_args:NNe 
        \prop_item:Nn
        \g_equation_prop
        { \int_use:N \g_version_int }
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% 6. Commande utilisateur
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\NewDocumentCommand{\Equation}{}
{
    \equation_show:
}


\ExplSyntaxOff

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{document}

\foreach \v in {1,...,3}
{
    \section*{Version \v}
    \SetVersion{\v}
    Résoudre l'équation suivante :
    \Equation
}

\end{document}
